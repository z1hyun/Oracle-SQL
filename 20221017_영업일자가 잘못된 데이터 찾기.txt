SELECT MST.SLRCAL_YMD  AS "기준영업일"
	 , MST.BFR_BIZN_YMD AS "이전영업일"
	 , MST.NEXT_BIZN_YMD AS "익일영업일"
	 , (SELECT MAX(SLRCAL_YMD) FROM YMD BFR WHERE MST.SLRCAL_YMD > BFR.SLRCAL_YMD AND HLDY_SECD = '1') AS "정상영업일(이전)" --이전 정상영업일
	 , (SELECT MIN(SLRCAL_YMD) FROM YMD NEX WHERE MST.SLRCAL_YMD < NEX.SLRCAL_YMD AND HLDY_SECD = '1') AS "정상영업일(이후)" --이후 정상영업일
FROM YMD MST 
WHERE MST.BFR_BIZN_YMD IN (SELECT SLRCAL_YMD FROM YMD WHERE HLDY_SECD != '1' AND MST.SLRCAL_YMD > SLRCAL_YMD) --영업일 이전 일자중에서 휴무 데이터
OR MST.NEXT_BIZN_YMD IN (SELECT SLRCAL_YMD FROM YMD WHERE HLDY_SECD != '1' AND MST.SLRCAL_YMD < SLRCAL_YMD) --영업일 이후 일자중에서 휴무 데이터
ORDER BY MST.SLRCAL_YMD
;
